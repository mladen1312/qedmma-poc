### Step 2: Vitis HLS for Performance-Critical IP

```cpp
// correlator_hls.cpp - High-Level Synthesis
#include <ap_fixed.h>
#include <hls_stream.h>

typedef ap_fixed<16,4> sample_t;
typedef ap_fixed<32,16> accum_t;

void zero_dsp_correlator(
    hls::stream<sample_t>& rx_in,
    hls::stream<accum_t>& range_out,
    const sample_t prbs_ref[32768],
    const int num_bins
) {
    #pragma HLS INTERFACE axis port=rx_in
    #pragma HLS INTERFACE axis port=range_out
    #pragma HLS INTERFACE s_axilite port=prbs_ref
    #pragma HLS INTERFACE s_axilite port=num_bins
    #pragma HLS INTERFACE s_axilite port=return
    
    #pragma HLS ARRAY_PARTITION variable=prbs_ref cyclic factor=64
    
    accum_t correlations[4096];
    #pragma HLS BIND_STORAGE variable=correlations type=RAM_2P impl=URAM
    
    // Parallel correlation
    for (int bin = 0; bin < num_bins; bin++) {
        #pragma HLS PIPELINE II=1
        #pragma HLS UNROLL factor=64
        
        sample_t sample = rx_in.read();
        
        for (int tap = 0; tap < 64; tap++) {
            correlations[bin] += sample * prbs_ref[bin + tap];
        }
    }
    
    // Output
    for (int i = 0; i < num_bins; i++) {
        #pragma HLS PIPELINE II=1
        range_out.write(correlations[i]);
    }
}
```
